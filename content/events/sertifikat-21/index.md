## Deskripsi Sertifikasi Kompetensi

Sertifikat ini diterbitkan untuk **Angga Sianipar** sebagai pengakuan atas keberhasilan menyelesaikan kursus **"Build an AI Agent"** yang disediakan oleh **IBM SkillsBuild**. Kredensial ini memvalidasi kemampuan teknis dalam memahami konsep, perancangan, dan implementasi agen kecerdasan buatan (AI Agent) yang cerdas dan fungsional.

---

### Kompetensi Inti yang Divalidasi
Pelatihan ini mencakup pemahaman mendalam mengenai pengembangan sistem AI, termasuk:
* **Konsep AI Agent**: Memahami arsitektur dasar dan fungsi dari agen kecerdasan buatan yang mampu berinteraksi dan menyelesaikan tugas secara otonom.
* **Pengembangan Solusi AI**: Praktik merancang agen untuk menyelesaikan tugas spesifik dalam ekosistem teknologi AI modern menggunakan standar industri.
* **Implementasi Teknologi IBM**: Memanfaatkan sumber daya pembelajaran dari IBM SkillsBuild untuk membangun dan mengoptimalkan solusi berbasis kecerdasan buatan.

  document.addEventListener("DOMContentLoaded", function() {
    // 1. Array berisi nama file gambar
    const imagesList = ["featured.jpg"];
    let currentIdx = 0;

    // 2. Cari target elemen gambar bawaan
    const themeImgTarget = document.querySelector('.article-header img, .featured-image img, img[class*="featured"]');
    
    if (themeImgTarget) {
      const themeContainer = themeImgTarget.parentElement;
      
      const sliderHTML = `
        <div class="cert-slider-container" style="position: relative; width: 100%; max-width: 800px; margin: 0 auto 20px auto; overflow: hidden; border-radius: 8px; box-shadow: 0 4px 20px rgba(0,0,0,0.3);">
          <div class="cert-slider-slides" style="display: flex; transition: transform 0.4s ease-in-out;">
            <img class="cert-slide-img" src="featured.jpg" style="width: 100%; flex-shrink: 0; cursor: zoom-in;" title="Klik untuk Mode Theater HD">
          </div>
          <button id="slider-prev" style="position: absolute; top: 50%; left: 15px; transform: translateY(-50%); background: rgba(0,0,0,0.6); color: white; border: none; width: 45px; height: 45px; border-radius: 50%; font-size: 24px; cursor: pointer; display: flex; align-items: center; justify-content: center; z-index: 10; user-select: none;">&#10094;</button>
          <button id="slider-next" style="position: absolute; top: 50%; right: 15px; transform: translateY(-50%); background: rgba(0,0,0,0.6); color: white; border: none; width: 45px; height: 45px; border-radius: 50%; font-size: 24px; cursor: pointer; display: flex; align-items: center; justify-content: center; z-index: 10; user-select: none;">&#10095;</button>
          <div id="slider-badge" style="position: absolute; bottom: 15px; right: 15px; background: rgba(0,0,0,0.7); color: #fff; padding: 4px 10px; border-radius: 12px; font-size: 12px; font-family: sans-serif; font-weight: bold; z-index: 10;">1 / 1</div>
        </div>
      `;
      
      themeContainer.innerHTML = sliderHTML;
    }

    const sliderWrapper = document.querySelector('.cert-slider-slides');
    const slideImages = document.querySelectorAll('.cert-slide-img');
    const sliderPrev = document.getElementById('slider-prev');
    const sliderNext = document.getElementById('slider-next');
    const sliderBadge = document.getElementById('slider-badge');
    const lightbox = document.getElementById('custom-light-box');
    const lightboxImg = document.getElementById('light-box-img');
    const closeBtn = document.getElementById('close-light-box');
    const prevBtn = document.getElementById('prev-light-box');
    const nextBtn = document.getElementById('next-light-box');

    function updateSlider(index) {
      currentIdx = index;
      if (sliderWrapper) sliderWrapper.style.transform = `translateX(-${currentIdx * 100}%)`;
      if (sliderBadge) sliderBadge.textContent = `${currentIdx + 1} / ${imagesList.length}`;
    }

    if (sliderPrev && sliderNext) {
      sliderPrev.addEventListener('click', function(e) {
        e.stopPropagation();
        let newIdx = currentIdx - 1;
        if (newIdx < 0) newIdx = imagesList.length - 1;
        updateSlider(newIdx);
      });

      sliderNext.addEventListener('click', function(e) {
        e.stopPropagation();
        let newIdx = currentIdx + 1;
        if (newIdx >= imagesList.length) newIdx = 0;
        updateSlider(newIdx);
      });
    }

    function updateLightboxImage(index) {
      currentIdx = index;
      lightboxImg.src = imagesList[currentIdx];
    }

    function openLightbox(index) {
      updateLightboxImage(index);
      lightbox.style.display = 'flex';
      setTimeout(() => { lightbox.style.opacity = '1'; }, 10);
      document.body.style.overflow = 'hidden';
    }

    const closeLightbox = function() {
      lightbox.style.opacity = '0';
      setTimeout(() => { 
        lightbox.style.display = 'none'; 
        document.body.style.overflow = ''; 
        updateSlider(currentIdx);
      }, 200);
    };

    slideImages.forEach((img, index) => {
      img.addEventListener('click', function() { openLightbox(index); });
    });

    prevBtn.addEventListener('click', function(e) {
      e.stopPropagation();
      let newIdx = currentIdx - 1;
      if (newIdx < 0) newIdx = imagesList.length - 1;
      updateLightboxImage(newIdx);
    });

    nextBtn.addEventListener('click', function(e) {
      e.stopPropagation();
      let newIdx = currentIdx + 1;
      if (newIdx >= imagesList.length) newIdx = 0;
      updateLightboxImage(newIdx);
    });

    closeBtn.addEventListener('click', closeLightbox);
    lightbox.addEventListener('click', function(e) {
      if (e.target === lightbox || e.target === lightboxImg) { closeLightbox(); }
    });

    document.addEventListener('keydown', function(e) {
      if (lightbox.style.display === 'flex') {
        if (e.key === "Escape") closeLightbox();
        if (e.key === "ArrowLeft") prevBtn.click();
        if (e.key === "ArrowRight") nextBtn.click();
      } else {
        if (e.key === "ArrowLeft" && sliderPrev) sliderPrev.click();
        if (e.key === "ArrowRight" && sliderNext) sliderNext.click();
      }
    });
  });
</script>
